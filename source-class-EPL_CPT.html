 127:     function __construct( $post_type_names, $options = array() ) {
 128: 
 129:         // Check if post type names is a string or an array.

 131:

 133:             $names = array(
 134:                 'singular',
 135:                 'plural',

 140:             $this->post_type_name = $post_type_names['post_type_name'];
 141: 
 142:             // Cycle through possible names.

 144:

 146:                 if ( isset( $post_type_names[ $name ] ) ) {
 147:

 150: 
 151:                 // Else generate the name.
 152:                 } else {
 153: 
 154:                     // define the method to be used
 155:                     $method = 'get_' . $name;
 156: 
 157:                     // Generate the name
 158:                     $this->$name = $this->$method();
 159:                 }
 160:             }
 161: 
 162:         // Else the post type name is only supplied.
 163:         } else {
 164: 
 165:             // Apply to post type name.
 166:             $this->post_type_name = $post_type_names;
 167: 
 168:             // Set the slug name.
 169:             $this->slug = $this->get_slug();
 170: 
 171:             // Set the plural name label.
 172:             $this->plural = $this->get_plural();
 173: 
 174:             // Set the singular name label.
 175:             $this->singular = $this->get_singular();
 176:             }
 177: 
 178:         // Set the user submitted options to the object.
 179:         $this->options = $options;
 180: 
 181:         // Register taxonomies.
 182:         $this->add_action( 'init', array( &$this, 'register_taxonomies' ) );
 183: 
 184:         // Register the post type.
 185:         $this->add_action( 'init', array( &$this, 'register_post_type' ) );
