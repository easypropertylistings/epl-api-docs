 20:  * @since  3.0
 21:  * @return void
 22: */
 23: function epl_contacts_page() {
 24:     $default_views = epl_contact_views();
 25:     $requested_view = isset( $_GET['view'] ) ? sanitize_text_field( $_GET['view'] ) : 'contacts';
 26:     if ( array_key_exists( $requested_view, $default_views ) && function_exists( $default_views[$requested_view] ) ) {
 27:         epl_render_contact_view( $requested_view, $default_views );
 28:     } else {
 29:         epl_contacts_list();
 30:     }
 31: }

 87: /**
 88:  * Renders the contact view wrapper
 89:  *
 90:  * @since  3.0
 91:  * @param  string $view      The View being requested
 92:  * @param  array $callbacks  The Registered views and their callback functions
 93:  * @return void
 94:  */
 95: function epl_render_contact_view( $view, $callbacks ) {
 96: 
 97:     $render = true;
 98: 
 99:     $contact_view_role = apply_filters( 'epl_view_contacts_role', 'manage_options' );
100: 
101:     if ( ! current_user_can( $contact_view_role ) ) {
102:         epl_set_error( 'epl-no-access', __( 'You are not permitted to view this data.', 'epl' ) );
103:         $render = false;
104:     }
105:     if($view == 'new-contact' || $view == 'all_tags') {
106:         $callbacks[$view]();
107:         return;
108:     }
109: 
110:     if ( ! isset( $_GET['id'] ) || ! is_numeric( $_GET['id'] ) ) {
111:         epl_set_error( 'epl-invalid_contact', __( 'Invalid Contact ID Provided.', 'epl' ) );
112:         $render = false;
113:     }
114: 
115:     $contact_id = (int)$_GET['id'];
116:     $contact    = new EPL_Contact( $contact_id );
117: 
118:     if ( empty( $contact->ID ) ) {
119:         epl_set_error( 'epl-invalid_contact', __( 'Invalid Contact ID Provided.', 'epl' ) );
120:         $render = false;
121:     }
122: 
123:     $contact_tabs = epl_contact_tabs();
124:     ?>
125: 
126:     <div class='wrap'>
127:         <h2><?php _e( 'Contact Details', 'epl' );?></h2>
128:         <?php if ( epl_get_errors() ) :?>
129:             <div class="error settings-error">
130:                 <?php epl_print_errors(); ?>
131:             </div>
132:         <?php endif; ?>
133: 
134:         <?php if ( $contact && $render ) : ?>
135: 
136:             <div id="epl-item-tab-wrapper" class="contact-tab-wrapper">
137:                 <ul id="epl-item-tab-wrapper-list" class="ustomer-tab-wrapper-list">
138:                 <?php foreach ( $contact_tabs as $key => $tab ) : ?>
139:                     <?php $active = $key === $view ? true : false; ?>
140:                     <?php $class  = $active ? 'active' : 'inactive'; ?>
141: 
142:                     <?php if ( ! $active ) : ?>
143:                     <a title="<?php echo esc_attr( $tab['title'] ); ?>" aria-label="<?php echo esc_attr( $tab['title'] ); ?>" href="<?php echo esc_url( admin_url( 'admin.php?page=epl-contacts&view=' . $key . '&id=' . $contact->id ) ); ?>">
144:                     <?php endif; ?>
145: 
146:                     <li class="<?php echo sanitize_html_class( $class ); ?>"><span class="dashicons <?php echo sanitize_html_class( $tab['dashicon'] ); ?>"></span></li>
147: 
148:                     <?php if ( ! $active ) : ?>
149:                     </a>
150:                     <?php endif; ?>
151: 
152:                 <?php endforeach; ?>
153:                 </ul>
154:             </div>
155: 
156:             <div id="epl-item-card-wrapper" class="epl-contact-card-wrapper" style="float: left">
157:                 <?php $callbacks[$view]( $contact ) ?>
158:             </div>
159: 
160:         <?php endif; ?>
161: 
162:     </div>
163: <?php
164: }
