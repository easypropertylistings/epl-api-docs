 25: 
 26:     Simple example:
 27: 
 28:     data format for each point: array( location on x, location on y )
 29: 
 30:     $data = array(
 31: 
 32:         'Label' => array(
 33:             array( 1, 5 ),
 34:             array( 3, 8 ),

 36:         ),
 37: 
 38:         'Second Label' => array(
 39:             array( 1, 7 ),
 40:             array( 4, 5 ),
 41:             array( 12, 8 )

 44: 
 45:     $graph = new EPL_Graph( $data );
 46:     $graph->display();
 47: 
 48:     */

 51:      * Data to graph
 52:      *
 53:      * @var array
 54:      * @since 3.0
 55:      */
 56:     private $data;
 57: 
 58:     /**

 61:      * @var string
 62:      * @since 3.0

 74:     /**
 75:      * Get things started
 76:      *
 77:      * @since 3.0
 78:      */
 79:     public function __construct( $_data ) {
 80: 
 81:         $this->data = $_data;
 82: 
 83:         // Generate unique ID
 84:         $this->id   = 'a' . md5( rand() );
 85: 
 86:         // Setup default options;
 87:         $this->options = array(
 88:             'y_mode'          => null,
 89:             'x_mode'          => null,
 90:             'y_decimals'      => 0,
 91:             'x_decimals'      => 0,
 92:             'y_position'      => 'right',
 93:             'time_format'     => '%d/%b',
 94:             'ticksize_unit'   => 'day',
 95:             'ticksize_num'    => 1,
 96:             'multiple_y_axes' => false,
 97:             'bgcolor'         => '#f9f9f9',
 98:             'bordercolor'     => '#ccc',
 99:             'color'           => '#bbb',
100:             'borderwidth'     => 2,
101:             'bars'            => false,
102:             'lines'           => true,
103:             'points'          => true
104:         );
105:     }
106: 
107:     /**
108:      * Set an option

116:     }

198:                     ],

213:                             timeFormat: "<?php echo $this->options['x_mode'] == 'time' ? $this->options['time_format'] : ''; ?>",
214:                             tickSize: "<?php echo $this->options['x_mode'] == 'time' ? '' : $this->options['ticksize_num']; ?>",

218:                         },
219:                         yaxis: {
