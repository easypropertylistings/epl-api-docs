 26:     function widget( $args, $instance ) {
 27:         $property_types = epl_get_active_post_types();
 28:         if ( ! empty( $property_types ) ) {
 29:             $property_types = array_keys( $property_types );
 30:         }
 31: 
 32:         $defaults = array(
 33:             'title'      => '',
 34:             'types'      => $property_types,
 35:             'featured'   => 0,
 36:             'status'     => array( 'current', 'sold', 'leased' ),
 37:             'view'       => 'default',
 38:             'display'    => 'image',
 39:             'image'      => 'thumbnail',
 40:             'archive'    => 0,
 41:             'order_rand' => 0,
 42:             'd_title'    => 0,
 43: 
 44:             'more_text'  => 'Read More',
 45:             'd_excerpt'  => 'off',
 46:             'd_suburb'   => 'on',
 47:             'd_street'   => 'on',
 48:             'd_price'    => 'on',
 49:             'd_more'     => 'on',
 50: 
 51:             'd_icons'    => 'none',
 52:             'p_number'   => 1,
 53:             'p_skip'     => 0,
 54:         );
 55:         $instance = wp_parse_args( (array) $instance, $defaults );
 56: 
 57:         if ( ! is_array( $instance['types'] ) ) {
 58:             $instance['types'] = array_map( 'trim', explode( ',', $instance['types'] ) );
 59:         }
 60:         if ( ! $instance['p_number'] ) {
 61:             $instance['p_number'] = 1;
 62:         }
 63:         if ( 'beds/baths' === $instance['d_icons'] ) {
 64:             $instance['d_icons'] = 'bb';
 65:         }
 66:         // WP_Query arguments.
 67:         $query_args = array(
 68:             'post_type'    => $instance['types'],
 69:             'showposts'    => absint( $instance['p_number'] ),
 70:             'offset'       => absint( $instance['p_skip'] ),
 71:             'paged'        => '1',
 72:             'epl_nopaging' => 'true',
 73:         );
 74: 
 75:         // Listing status.
 76:         if ( ! empty( $instance['status'] ) ) {
 77:             if ( ! is_array( $instance['status'] ) ) {
 78:                 $instance['status'] = array_map( 'trim', explode( ',', $instance['status'] ) );
 79:             }
 80:             // If property_status is any do not set meta_query for property_status.
 81:             if ( ! in_array( 'any', $instance['status'] ) ) {
 82:                 $query_args['meta_query'][] = array(
 83:                     'key'       => 'property_status',
 84:                     'value'     => $instance['status'],
 85:                     'compare'   => 'IN',
 86:                 );
 87:             }
 88:         }
 89:         // Loading featured listings.
 90:         if ( 'on' === $instance['featured'] ) {
 91:             $query_args['meta_query'][] = array(
 92:                 'key'   => 'property_featured',
 93:                 'value' => 'yes',
 94:             );
 95:         }
 96:         // Loading current archive page listings.
 97:         if ( 'on' === $instance['archive'] && is_post_type_archive() ) {
 98:             $get_current_type = get_post_type();
 99:             if ( false !== $get_current_type && in_array( $get_current_type , $property_types ) ) {
100:                 $query_args['post_type'] = $get_current_type;
101:             }
102:         }
103:         // Using random order for listings.
104:         if ( 'on' === $instance['order_rand'] ) {
105:             $query_args['orderby'] = 'rand';
106:         }
107: 
108:         $query = new WP_Query( $query_args );
109:         epl_get_template_part( 'widgets/listing/' . ( strlen( trim( $instance['view'] ) ) ? trim( $instance['view'] ) : 'default' ) . '.php',
110:             array(
111:                 'args'     => $args,
112:                 'instance' => $instance,
113:                 'query'    => $query,
114:             )
115:         );
116:     }

118:     function update( $new_instance, $old_instance ) {
119:         $instance = $old_instance;
120:         $instance['title']      = strip_tags( $new_instance['title'] );
121:         // Saving listing type.
122:         $epl_post_types         = epl_get_active_post_types();
123:         if ( isset( $new_instance['types'] ) && is_array( $new_instance['types'] ) ) {
124:             $types = array();
125:             foreach ( $new_instance['types'] as $type ) {
126:                 if ( array_key_exists( $type, $epl_post_types ) ) {
127:                     $types[] = $type;
128:                 }
129:             }
130:             if ( count( $types ) ) {
131:                 $instance['types'] = $types;
132:             }
133:         }
134:         // Saving listing status.
135:         $listing_statuses = apply_filters( 'epl_widget_listing_property_status',
136:             array(
137:                 'current'   => __( 'Current', 'epl' ),
138:                 'sold'      => __( 'Sold', 'epl' ),
139:                 'leased'    => __( 'Leased', 'epl' ),
140:                 'withdrawn' => __( 'Withdrawn', 'epl' ),
141:                 'offmarket' => __( 'Off Market', 'epl' ),
142:             )
143:         );
144:         if ( isset( $new_instance['status'] ) && is_array( $new_instance['status'] ) ) {
145:             $statuses = array();
146:             foreach ( $new_instance['status'] as $status ) {
147:                 if ( array_key_exists( $status, $listing_statuses ) ) {
148:                     $statuses[] = $status;
149:                 }
150:             }
151:             if ( count( $statuses ) ) {
152:                 $instance['status'] = $statuses;
153:             }
154:         }
155:         $instance['featured']   = strip_tags( $new_instance['featured'] );
156:         $instance['view']       = strip_tags( $new_instance['view'] );
157:         $instance['display']    = strip_tags( $new_instance['display'] );
158:         $instance['image']      = strip_tags( $new_instance['image'] );
159:         $instance['archive']    = strip_tags( $new_instance['archive'] );
160:         $instance['d_title']    = strip_tags( $new_instance['d_title'] );
161: 
162:         $instance['more_text']  = strip_tags( $new_instance['more_text'] );
163:         $instance['d_excerpt']  = strip_tags( $new_instance['d_excerpt'] );
164:         $instance['d_suburb']   = strip_tags( $new_instance['d_suburb'] );
165:         $instance['d_street']   = strip_tags( $new_instance['d_street'] );
166:         $instance['d_price']    = strip_tags( $new_instance['d_price'] );
167:         $instance['d_more']     = strip_tags( $new_instance['d_more'] );
168: 
169:         $instance['d_icons']    = strip_tags( $new_instance['d_icons'] );
170:         $instance['p_number']   = strip_tags( $new_instance['p_number'] );
171:         $instance['p_skip']     = strip_tags( $new_instance['p_skip'] );
